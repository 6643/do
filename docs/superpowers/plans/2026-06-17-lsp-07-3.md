# do lsp 07.3 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a first-version `do lsp` command that serves diagnostics over LSP stdio without adding completion, hover, definition, or formatting.

**Architecture:** Keep the language server under `tool/lsp/` and reuse the existing compiler front-end stages for diagnostics. First expose structured diagnostics from `tool/build/diag.zig`, then build a pure diagnostics collector that never prints or exits, then wrap it with a minimal JSON-RPC stdio server. The first server publishes at most the current compiler's first diagnostic per document because the parser/sema/import chain currently fail fast.

**Tech Stack:** Zig compiler code, LSP JSON-RPC over stdio, Node-based smoke harness, shell regression suite, Markdown docs.

```zig
pub const LspArgs = struct {
    stdio: bool = true,
};

pub fn parseLsp(args: []const []const u8) !LspArgs {
    var saw_stdio = false;
    var i: usize = 1;
    while (i < args.len) : (i += 1) {
        if (std.mem.eql(u8, args[i], "--stdio")) {
            if (saw_stdio) return error.UnexpectedCliArg;
            saw_stdio = true;
            continue;
        }
        return error.UnexpectedCliArg;
    }
    return .{ .stdio = true };
}
```

- [ ] **Step 3: Implement fail-fast diagnostics collection**

Implement this public API:

```zig
pub const Diagnostic = build_diag.CompileDiagnostic;

pub fn collectDiagnostics(
    allocator: std.mem.Allocator,
    path: []const u8,
    source: []const u8,
    dep_root: []const u8,
) ![]Diagnostic {
    const tokens = lexer.tokenize(allocator, source) catch |err| {
        return one(allocator, build_diag.buildCompileDiagnostic(path, source, null, err, null));
    };
    defer allocator.free(tokens);

    var program = parser.parseProgram(allocator, tokens, source.len) catch |err| {
        return one(allocator, build_diag.buildCompileDiagnostic(path, source, tokens, err, parserErrorLoc()));
    };
    defer program.deinit(allocator);

    sema.checkProgram(allocator, program, tokens) catch |err| {
        return one(allocator, build_diag.buildCompileDiagnostic(path, source, tokens, err, semaErrorLoc()));
    };

    var graph = imports.checkAndLoad(std.Io.null, allocator, path, tokens, dep_root) catch |err| {
        return one(allocator, build_diag.buildCompileDiagnostic(path, source, tokens, err, importsErrorLoc()));
    };
    graph.deinit();

    return allocator.alloc(Diagnostic, 0);
}
```

Required helper behavior:

- `one(...)` returns an owned one-element slice.
- `parserErrorLoc()`, `semaErrorLoc()`, and `importsErrorLoc()` mirror the private helpers currently in `tool/build/run.zig`.
- No code path calls `std.process.exit`.
- No code path prints diagnostics to stderr.

```zig
    uri: []const u8,
    source: []const u8,
};

pub const ServerState = struct {
    allocator: std.mem.Allocator,
    dep_root: []const u8,
    documents: std.ArrayList(Document),
    shutdown_requested: bool = false,

    pub fn init(allocator: std.mem.Allocator, dep_root: []const u8) ServerState
    pub fn deinit(self: *ServerState) void
    pub fn didOpen(self: *ServerState, uri: []const u8, text: []const u8) !void
    pub fn didChange(self: *ServerState, uri: []const u8, text: []const u8) !void
    pub fn didClose(self: *ServerState, uri: []const u8) !void
    pub fn findDocument(self: *const ServerState, uri: []const u8) ?*const Document
};
```

Implement `pub fn run(init: std.process.Init, args: []const []const u8) !void`:

- Parse args with `cli.parseLsp(args)`.
- Resolve dependency root with the same rule as build/test: `DO_LIB_ROOT` or `$HOME/.do/lib`.
- Read JSON-RPC frames from stdin.
- Handle:
  - `initialize`: respond with diagnostics-only capabilities.
  - `initialized`: no response.
  - `textDocument/didOpen`: store source and publish diagnostics.
  - `textDocument/didChange`: replace whole source from `contentChanges[0].text` and publish diagnostics.
  - `textDocument/didClose`: remove document and publish empty diagnostics.
  - `shutdown`: respond and set `shutdown_requested`.
  - `exit`: return from the loop.
- For unknown requests with an `id`, respond with JSON-RPC error `-32601`.
- For unknown notifications without `id`, ignore.

```js
import { readFile } from "node:fs/promises";
import { spawn } from "node:child_process";

function frame(message) {
  const body = JSON.stringify(message);
  return `Content-Length: ${Buffer.byteLength(body, "utf8")}\r\n\r\n${body}`;
}

const [doBin, fixturePath] = process.argv.slice(2);
const fixture = JSON.parse(await readFile(fixturePath, "utf8"));
const child = spawn(doBin, ["lsp"], {
  stdio: ["pipe", "pipe", "pipe"],
  env: { ...process.env, DO_LIB_ROOT: "tool/build/test/lib" },
});

let stdout = "";
let stderr = "";
child.stdout.setEncoding("utf8");
child.stderr.setEncoding("utf8");
child.stdout.on("data", (chunk) => { stdout += chunk; });
child.stderr.on("data", (chunk) => { stderr += chunk; });

for (const message of fixture.messages) {
  child.stdin.write(frame(message));
}
child.stdin.end();

const exitCode = await new Promise((resolve) => {
  child.on("close", resolve);
});

if (exitCode !== 0) {
  console.error(stderr);
  process.exit(1);
}

for (const needle of fixture.expect) {
  if (!stdout.includes(needle)) {
    console.error(`missing expected output: ${needle}`);
    console.error(stdout);
    process.exit(1);
  }
}

console.log(`ok: lsp ${fixture.name}`);
```
